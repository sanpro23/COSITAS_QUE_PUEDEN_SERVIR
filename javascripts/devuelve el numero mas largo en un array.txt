function largestOfFour(arr) {
  // 1. Crea una matriz vacía para guardar los resultados.
  const largestNumbers = [];

  // 2. Itera sobre cada sub-matriz de la matriz principal (arr).
  for (let i = 0; i < arr.length; i++) {
    // 3. Dentro de cada sub-matriz, encuentra el número más grande.
    //    Puedes usar el método `Math.max()` para esto.
    //    El operador de propagación (...) expande los elementos de la sub-matriz.
    const max = Math.max(...arr[i]);

    // 4. Agrega el número más grande encontrado a la matriz de resultados.
    largestNumbers.push(max);
  }

  // 5. Devuelve la nueva matriz con los números más grandes.
  return largestNumbers;
}

largestOfFour([
  [4, 5, 1, 3],
  [13, 27, 18, 26],
  [32, 35, 37, 39],
  [1000, 1001, 857, 1]
]);
