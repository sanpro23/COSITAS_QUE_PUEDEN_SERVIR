function titleCase(str) {
  let palabras = str.toLowerCase().split(" ");
 for (let i = 0; i < palabras.length; i++) {
   palabras[i] = palabras[i][0].toUpperCase() + palabras[i].slice(1);
 }return palabras.join(" ");

}

titleCase("I'm a little tea pot");
