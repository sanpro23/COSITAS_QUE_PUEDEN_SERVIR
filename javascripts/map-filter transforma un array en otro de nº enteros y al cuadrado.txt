const squareList = arr => {
  return arr
    .filter(num => num > 0 && Number.isInteger(num)) // solo enteros positivos
    .map(num => num ** 2); // elevar al cuadrado
};

const squaredIntegers = squareList([-3, 4.8, 5, 3, -3.2]);
console.log(squaredIntegers);
