function sumAll(arr) {
  const min = Math.min(...arr); // Encuentra el menor
  const max = Math.max(...arr); // Encuentra el mayor
  let total = 0;
  for (let i = min; i <= max; i++) { //recorre desde el menor hasta el mayor.
    total += i; //va sumando cada nÃºmero.
  }
  return total;
}
console.log(sumAll([4, 1]));
sumAll([1, 4]);
