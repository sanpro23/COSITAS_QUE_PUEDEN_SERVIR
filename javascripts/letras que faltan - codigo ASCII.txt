function fearNotLetter(str) {
  
  for (let i = 0; i < str.length - 1; i++) { // Recorremos la cadena letra por letra
    
    const codigoActual = str.charCodeAt(i); // Código ASCII de la letra actual
    
    const codigoSiguiente = str.charCodeAt(i + 1); // Código ASCII de la siguiente letra

    if (codigoSiguiente - codigoActual > 1) {  // Si la diferencia no es 1, falta una letra en medio
      
      return String.fromCharCode(codigoActual + 1); // Devolvemos la letra que falta usando String.fromCharCode
    }
  }
  return undefined;  // Si no falta ninguna letra, devolvemos undefined
}
