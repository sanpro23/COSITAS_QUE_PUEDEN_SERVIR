function telephoneCheck(str) {
  // Expresión regular que contempla los formatos válidos
   const regex = /^(1\s?)?(\(\d{3}\)|\d{3})([\s\-]?)\d{3}([\s\-]?)\d{4}$/;
     
      return regex.test(str);  // Devuelve true si la cadena coincide con el patrón
    }

    function validar() {
      const numero = document.getElementById("telefono").value; // Obtiene el valor ingresado  de texto
      const esValido = telephoneCheck(numero); // Llama a la función de validación
      const resultado = document.getElementById("resultado"); // donde se mostrará el resultado
      
      if (esValido) { // Muestra el mensaje correspondiente según el resultado
        resultado.textContent = "✅ Número válido de EE. UU.";
        resultado.style.color = "green"; // Cambia el color del texto a verde
      } else {
        resultado.textContent = "❌ Número inválido.";
        resultado.style.color = "red"; // Cambia el color del texto a rojo
      }
    }
